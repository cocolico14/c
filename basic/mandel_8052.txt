1 REM Mandelbrot
2 PY=1
3 PX=1
4 X=0
5 XT=0
6 Y=0
7 I=0
8 XT = (X*X)/10 - (Y*Y)/10 + (PX-23)
9 Y = (X*Y)/5 + (10*PY - 75)/8
10 X = XT
11 IF (X/10)*X + (Y/10)*Y >= 400 THEN GOTO 15
12 I=I+1:IF I<11 THEN GOTO 8
13 PRINT "X",
14 GOTO 16
15 PRINT " ",
16 PX=PX+1:IF PX<31 THEN GOTO 4
17 PRINT
18 PY=PY+1:IF PY<15 THEN GOTO 3
19 END
